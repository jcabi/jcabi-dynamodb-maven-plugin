103     /**
104      * Public ctor.
105      * @param locations The location of the tables to be created, in JSON format
106      * @param endpoint AWS endpoint
107      * @param port Tcp port
108      * @param key AWS key
109      * @param secret AWS secret
110      * @checkstyle ParameterNumberCheck (3 lines)
111      */
112     public Tables(final Collection<String> locations, final String endpoint,
113         final Integer port, final String key, final String secret) {
114         this.locations = locations;
115         this.endpoint = endpoint;
116         this.port = port;
117         this.key = key;
118         this.secret = secret;
119     }

121     /**
122      * Creates tables.
123      * @throws IOException if something goes wrong
124      */
125     public void create() throws IOException {
126         final AmazonDynamoDB aws = new AmazonDynamoDBClient(
127             new BasicAWSCredentials(this.key, this.secret)
128         );
129         aws.setEndpoint(String.format("%s:%d", this.endpoint, this.port));
130         for (final String table : this.locations) {
131             final JsonObject json = this.readJson(table);
132             if (json.containsKey("TableName")) {
133                 final String name = json.getString("TableName");
134                 if (Tables.exists(aws, name)) {
135                     Logger.info(
136                         this, "Table '%s' already exists, skipping...", name
137                     );
138                 } else {
139                     this.createTable(aws, json);
140                 }
141             } else {
142                 throw new IOException(
143                     String.format(
144                         "File '%s' does not specify TableName attribute", table
145                     )
146                 );
147             }
148         }
149     }

151     /**
152      * Table exists?
153      * @param aws AWS
154      * @param name Table name
155      * @return TRUE if it exists
156      */
157     private static boolean exists(final AmazonDynamoDB aws, final String name) {
158         boolean exists;
159         try {
160             TableUtils.waitUntilExists(
161                 aws, name, Tv.THOUSAND, Tv.HUNDRED
162             );
163             exists = true;
164         } catch (final AmazonClientException ex) {
165             exists = false;
166         } catch (final InterruptedException ex) {
167             Thread.currentThread().interrupt();
168             throw new IllegalStateException(ex);
169         }
170         return exists;
171     }

173     /**
174      * Create DynamoDB table.
175      * @param aws DynamoDB client
176      * @param json JSON definition of table
177      * @checkstyle ExecutableStatementCount (50 lines)
178      * @checkstyle NPathComplexityCheck (100 lines)
179      */
180     private void createTable(final AmazonDynamoDB aws, final JsonObject json) {
181         final String name = json.getString("TableName");
182         final CreateTableRequest request =
183             new CreateTableRequest().withTableName(name);
184         if (json.containsKey("KeySchema")) {
185             final Collection<KeySchemaElement> keys = this.keySchema(json);
186             request.setKeySchema(keys);
187         }
188         if (json.containsKey("AttributeDefinitions")) {
189             final Collection<AttributeDefinition> attrs =
190                 new LinkedList<AttributeDefinition>();
191             final JsonArray schema =
192                 json.getJsonArray("AttributeDefinitions");
193             for (final JsonObject defn : schema.getValuesAs(JsonObject.class)) {
194                 attrs.add(
195                     new AttributeDefinition(
196                         defn.getString("AttributeName"),
197                         defn.getString("AttributeType")
198                     )
199                 );
200             }
201             request.setAttributeDefinitions(attrs);
202         }
203         if (json.containsKey("ProvisionedThroughput")) {
204             final JsonObject throughput =
205                 json.getJsonObject("ProvisionedThroughput");
206             request.setProvisionedThroughput(
207                 new ProvisionedThroughput(
208                     this.asLong(throughput, "ReadCapacityUnits"),
209                     this.asLong(throughput, "WriteCapacityUnits")
210                 )
211             );
212         }
213         if (json.containsKey("GlobalSecondaryIndexes")) {
214             final Collection<GlobalSecondaryIndex> indexes =
215                 new LinkedList<GlobalSecondaryIndex>();
216             final JsonArray array =
217                 json.getJsonArray("GlobalSecondaryIndexes");
218             for (final JsonObject index : array.getValuesAs(JsonObject.class)) {
219                 final JsonObject throughput =
220                     index.getJsonObject("ProvisionedThroughput");
221                 final GlobalSecondaryIndex gsi = new GlobalSecondaryIndex()
222                     .withIndexName(index.getString("IndexName"))
223                     .withKeySchema(this.keySchema(index))
224                     .withProjection(this.projection(index))
225                     .withProvisionedThroughput(
226                         new ProvisionedThroughput(
227                             this.asLong(throughput, "ReadCapacityUnits"),
228                             this.asLong(throughput, "WriteCapacityUnits")
229                         )
230                     );
231                 indexes.add(gsi);
232             }
233             request.setGlobalSecondaryIndexes(indexes);
234         }
235         if (json.containsKey("LocalSecondaryIndexes")) {
236             final Collection<LocalSecondaryIndex> indexes =
237                 new LinkedList<LocalSecondaryIndex>();
238             final JsonArray array =
239                 json.getJsonArray("LocalSecondaryIndexes");
240             for (final JsonObject index : array.getValuesAs(JsonObject.class)) {
241                 final LocalSecondaryIndex lsi = new LocalSecondaryIndex()
242                     .withIndexName(index.getString("IndexName"))
243                     .withKeySchema(this.keySchema(index))
244                     .withProjection(this.projection(index));
245                 indexes.add(lsi);
246             }
247             request.setLocalSecondaryIndexes(indexes);
248         }
249         aws.createTable(request);
250         Logger.info(this, "Waiting for table '%s' to become active", name);
251         try {
252             TableUtils.waitUntilActive(aws, name);
253         } catch (final InterruptedException ex) {
254             Thread.currentThread().interrupt();
255             throw new IllegalStateException(ex);
256         }
257         Logger.info(this, "Table '%s' is now ready for use", name);
258     }

260     /**
261      * Get json value as a long - with a backward compatibility fallback for
262      * string values.
263      * @param json JSON input
264      * @param name The element name
265      * @return The element value converted as a long
266      */
267     private long asLong(final JsonObject json, final String name) {
268         long result;
269         try {
270             result = json.getJsonNumber(name).longValue();
271         } catch (final ClassCastException ex) {
272             result = Long.parseLong(json.getString(name));
273         }
274         return result;
275     }

277     /**
278      * Get projection JSON element.
279      * @param json JSON input
280      * @return Projection
281      */
282     private Projection projection(final JsonObject json) {
283         final JsonObject projn = json.getJsonObject("Projection");
284         final Projection projection = new Projection()
285             .withProjectionType(projn.getString("ProjectionType"));
286         final Collection<String> nonkeyattrs = new LinkedList<String>();
287         if (projn.containsKey("NonKeyAttributes")) {
288             for (final JsonValue nonkey
289                 : projn.getJsonArray("NonKeyAttributes")) {
290                 nonkeyattrs.add(nonkey.toString());
291             }
292             projection.setNonKeyAttributes(nonkeyattrs);
293         }
294         return projection;
295     }
296 
297     /**
298      * Get key schema elements.
299      * @param json JSON input
300      * @return Key schema elements
301      */
302     private Collection<KeySchemaElement> keySchema(final JsonObject json) {
303         final Collection<KeySchemaElement> keys =
304             new LinkedList<KeySchemaElement>();
305         final JsonArray schema = json.getJsonArray("KeySchema");
306         for (final JsonValue value : schema) {
307             final JsonObject element = (JsonObject) value;
308             keys.add(
309                 new KeySchemaElement(
310                     element.getString("AttributeName"),
311                     element.getString("KeyType")
312                 )
313             );
314         }
315         return keys;
316     }

318     /**
319      * Reads a file's contents into a JsonObject.
320      * @param file The path of the file to read
321      * @return The JSON object
322      * @throws IOException If there is an execution failure.
323      */
324     private JsonObject readJson(final String file) throws IOException {
325         InputStream stream = null;
326         JsonObject json = null;
327         try {
328             stream = new FileInputStream(file);
329             json = Json.createReader(stream).readObject();
330         } catch (final FileNotFoundException ex) {
331             throw new IOException("Failed to read table definition", ex);
332         } finally {
333             if (stream != null) {
334                 try {
335                     stream.close();
336                 } catch (final IOException ex) {
337                     Logger.error(
338                         this,
339                         "Failed to close stream with message %s",
340                         ex.getMessage()
341                     );
342                 }
343             }
344         }
345         return json;
346     }
